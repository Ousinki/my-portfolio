.my-button {
    background-color: #007bff;
    color: white;
    padding: 10px 20px;
    border: none;
    border-radius: 5px;
    cursor: pointer;
    transition: all 0.3s;
}

.my-button:hover {
    background-color: #0056b3;
    transform: scale(1.05);
}
